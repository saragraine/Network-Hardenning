## client VM

cd ~/TD4

curl -vk https://10.10.20.10:8443/ 2>&1 | tee evidence/before/curl_vk.txt
curl -skI https://10.10.20.10:8443/ | tee evidence/before/headers.txt

openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1 </dev/null | tee evidence/before/openssl_tls1.txt
openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1_1 </dev/null | tee evidence/before/openssl_tls1_1.txt
openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1_2 </dev/null | tee evidence/before/openssl_tls1_2.txt

./testssl.sh --fast --warnings batch https://10.10.20.10:8443 | tee evidence/before/tls_scan.txt

curl -vk https://10.10.20.10:8443/ 2>&1 | tee evidence/after/curl_vk.txt
curl -skI https://10.10.20.10:8443/ | tee evidence/after/headers.txt

openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1 </dev/null | tee evidence/after/openssl_tls1.txt
openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1_1 </dev/null | tee evidence/after/openssl_tls1_1.txt
openssl s_client -connect 10.10.20.10:8443 -servername td4.local -tls1_2 </dev/null | tee evidence/after/openssl_tls1_2.txt
openssl s_client -connect 10.10.20.10:8443 -servername td4.local </dev/null | tee evidence/after/openssl_s_client.txt

./testssl.sh --fast --warnings batch https://10.10.20.10:8443 | tee evidence/after/tls_scan.txt

for i in $(seq 1 30); do curl -sk -o /dev/null -w "%{http_code}\n" https://10.10.20.10:8443/api; done | tee evidence/after/rate_limit_test.txt

curl -k https://10.10.20.10:8443/

## srv-web VM

sudo apt update
sudo apt install -y nginx openssl curl tcpdump

sudo mkdir -p /etc/nginx/certs
cd /etc/nginx/certs

sudo openssl req -x509 -nodes -days 7 \
-newkey rsa:2048 \
-keyout server.key \
-out server.crt \
-subj "/CN=td4.local"

sudo openssl x509 -in /etc/nginx/certs/server.crt -noout -text | tee /tmp/cert_info_before.txt

sudo mkdir -p /var/www/td4
echo '<h1>TD4 TLS Lab</h1><p>Weak baseline.</p>' | sudo tee /var/www/td4/index.html

sudo tee /etc/nginx/sites-available/td4 > /dev/null <<'EOF'
server {
    listen 8443 ssl;
    server_name td4.local;

    ssl_certificate     /etc/nginx/certs/server.crt;
    ssl_certificate_key /etc/nginx/certs/server.key;

    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_ciphers HIGH:MEDIUM:!aNULL;
    ssl_prefer_server_ciphers on;

    root /var/www/td4;
    index index.html;

    location / {
        try_files $uri $uri/ =404;
    }
}
EOF

sudo ln -s /etc/nginx/sites-available/td4 /etc/nginx/sites-enabled/td4
sudo rm -f /etc/nginx/sites-enabled/default
sudo nginx -t
sudo systemctl reload nginx

curl -k https://127.0.0.1:8443/

sudo tee /etc/nginx/sites-available/td4 > /dev/null <<'EOF'
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;

server {
    listen 8443 ssl;
    server_name td4.local;

    ssl_certificate     /etc/nginx/certs/server.crt;
    ssl_certificate_key /etc/nginx/certs/server.key;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE+AESGCM:ECDHE+CHACHA20:!aNULL:!MD5:!RC4;
    ssl_prefer_server_ciphers on;

    add_header Strict-Transport-Security "max-age=300" always;

    if ($http_user_agent ~* "sqlmap|nikto|dirbuster") {
        return 403;
    }

    root /var/www/td4;
    index index.html;

    location / {
        try_files $uri $uri/ =404;
    }

    location /api {
        limit_req zone=api_limit burst=5 nodelay;
        return 200 "API response\n";
    }
}
EOF

sudo nginx -t
sudo systemctl reload nginx

sudo openssl x509 -in /etc/nginx/certs/server.crt -noout -text | tee /tmp/cert_info_after.txt

ip -br addr
sudo tcpdump -i <iface> -w /tmp/tls_handshake_after.pcap 'host 10.10.20.10 and tcp port 8443' -c 50

sudo tail -n 100 /var/log/nginx/access.log | tee /tmp/nginx_access_after.txt
sudo tail -n 100 /var/log/nginx/error.log | tee /tmp/nginx_error_after.txt
